import weaviate
import json 

where_filter = {
    "path": ["name"],
    "operator": "Like",
    "valueString": "New York*"
  }

client = weaviate.Client('https://some-endpoint.semi.network')
query_result = (
  client.query
  .get("Publication", "name")
  .with_where(where_filter)
  .do()
)

print(json.dumps(query_result, indent=4))

<TabItem value="js" label="JavaScript">

```js
weaviate = require("weaviate-client");

const client = weaviate.client({
    scheme: "https",
    host: "some-endpoint.semi.network"
});

const where_filter = {
    "path": ["name"],
    "operator": "Like",
    "valueString": "New York*"
}

client.graphql
    .get()
    .withClassName('Publication')
    .withFields('name')
    .withWhere(where_filter)
    .do()
    .then(res => {
        console.log(res)
    })
    .catch(err => {
        console.error(err)
    });
```
